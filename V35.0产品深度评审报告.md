### 2.1 P0安全问题（必须立即修复）

#### 问题1：密码哈希算法安全性不足 ⚠️

**问题位置**：`business-common/member_service.py:196-198`

```python
# 当前实现 - 使用简单SHA256，无salt
def _hash_password(self, password: str) -> str:
    return hashlib.sha256(password.encode()).hexdigest()
```

**风险分析**：
- SHA256无盐哈希，易被彩虹表攻击
- 无密码强度验证
- 无登录失败锁定机制

**真实场景问题**：
- 用户使用弱密码（如"123456"）可能被快速破解
- 数据库泄露后，攻击者可批量破解用户密码

**修复建议**：
```python
# 推荐使用bcrypt
import bcrypt

def _hash_password(self, password: str) -> str:
    salt = bcrypt.gensalt(rounds=12)
    return bcrypt.hashpw(password.encode(), salt).decode()

def verify_password(self, password: str, hashed: str) -> bool:
    return bcrypt.checkpw(password.encode(), hashed.encode())
```

#### 问题3：库存扣减存在并发问题 ⚠️

**问题位置**：`business-common/order_service.py:247-261`

```python
def _lock_stock(self, sku_id: int, quantity: int, is_product: bool = False):
    if is_product:
        sql = """
            UPDATE business_products
            SET stock = stock - %s
            WHERE id = %s AND stock >= %s
        """
    else:
        sql = """
            UPDATE business_product_skus
            SET stock = stock - %s
            WHERE id = %s AND stock >= %s
        """
    db.execute(sql, [quantity, sku_id, quantity])
```

**风险分析**：
- 虽然有 `stock >= %s` 条件，但高并发下可能超卖
- 无数据库行锁保护
- 事务边界不清晰

**真实场景问题**：
- 秒杀活动时，大量用户同时下单可能超卖
- 超卖后用户支付成功但无货可发

**修复建议**：
```python
# 使用悲观锁
def _lock_stock_with_lock(self, sku_id: int, quantity: int):
    conn = db.get_db()
    cursor = conn.cursor()
    try:
        # 使用 SELECT ... FOR UPDATE 获取行锁
        cursor.execute(
            "SELECT stock FROM business_product_skus WHERE id = %s FOR UPDATE",
            [sku_id]
        )
        result = cursor.fetchone()
        if result['stock'] < quantity:
            raise Exception("库存不足")
        
        cursor.execute(
            "UPDATE business_product_skus SET stock = stock - %s WHERE id = %s",
            [quantity, sku_id]
        )
        conn.commit()
    except:
        conn.rollback()
        raise
    finally:
        cursor.close()
        conn.close()
```

# member_service.py

def _hash_password(self, password: str, version: int = 2) -> str:
    """密码哈希 - V35.0支持多版本"""
    if version == 1:
        # 兼容旧版
        return hashlib.sha256(password.encode()).hexdigest()
    elif version == 2:
        # bcrypt
        import bcrypt
        salt = bcrypt.gensalt(rounds=12)
        return bcrypt.hashpw(password.encode(), salt).decode()
    raise ValueError(f"不支持的密码版本: {version}")

def verify_password(self, password: str, hashed: str) -> bool:
    """验证密码"""
    if hashed.startswith('$2'):  # bcrypt格式
        import bcrypt
        return bcrypt.checkpw(password.encode(), hashed.encode())
    else:
        # 兼容旧SHA256
        return hashlib.sha256(password.encode()).hexdigest() == hashed

def _check_account_locked(self, member: dict) -> bool:
    """检查账户是否被锁定"""
    if member.get('lock_until'):
        if datetime.now() < member['lock_until']:
            return True
    return False

def _handle_login_fail(self, user_id: int):
    """处理登录失败"""
    fail_count = db.get_one(
        "SELECT login_fail_count FROM business_members WHERE user_id=%s",
        [user_id]
    )['login_fail_count'] + 1
    
    lock_until = None
    if fail_count >= 5:
        lock_until = datetime.now() + timedelta(minutes=30)
    
    db.execute("""
        UPDATE business_members 
        SET login_fail_count=%s, lock_until=%s 
        WHERE user_id=%s
    """, [fail_count, lock_until, user_id])

### 5.2 Token安全升级

**方案选择**：Redis + JWT混合方案

```python
# auth.py

import jwt
import redis
from datetime import datetime, timedelta

# Redis连接
redis_client = redis.Redis(host='localhost', port=6379, db=0)
JWT_SECRET = os.environ.get('JWT_SECRET', 'your-secret-key')

class TokenManager:
    """Token管理器 - V35.0"""
    
    @staticmethod
    def generate_token(user_id: int, token_type: str = 'access') -> str:
        """生成Token"""
        now = datetime.utcnow()
        if token_type == 'access':
            exp = now + timedelta(hours=2)
        else:
            exp = now + timedelta(days=7)
        
        payload = {
            'user_id': user_id,
            'type': token_type,
            'iat': now,
            'exp': exp,
            'jti': str(uuid.uuid4())  # Token唯一ID
        }
        
        token = jwt.encode(payload, JWT_SECRET, algorithm='HS256')
        
        # 存储到Redis（用于主动失效）
        redis_client.setex(
            f"token:{payload['jti']}",
            int((exp - now).total_seconds()),
            json.dumps({'user_id': user_id, 'type': token_type})
        )
        
        return token
    
    @staticmethod
    def verify_token(token: str) -> dict:
        """验证Token"""
        try:
            payload = jwt.decode(token, JWT_SECRET, algorithms=['HS256'])
            
            # 检查Redis中是否存在
            jti = payload.get('jti')
            if jti and not redis_client.exists(f"token:{jti}"):
                return None  # Token已失效
            
            return payload
        except jwt.ExpiredSignatureError:
            return None  # Token过期
        except jwt.InvalidTokenError:
            return None  # Token无效
    
    @staticmethod
    def revoke_token(token: str) -> bool:
        """主动失效Token"""
        try:
            payload = jwt.decode(token, JWT_SECRET, algorithms=['HS256'], options={"verify_exp": False})
            jti = payload.get('jti')
            if jti:
                redis_client.delete(f"token:{jti}")
            return True
        except:
            return False
```

### 5.3 库存并发优化

```python
# order_service.py

def _lock_stock_with_pessimistic_lock(self, sku_id: int, quantity: int, 
                                      is_product: bool = False) -> bool:
    """使用悲观锁扣减库存 - V35.0"""
    conn = db.get_db()
    cursor = conn.cursor()
    
    try:
        conn.begin()
        
        table = 'business_product_skus' if not is_product else 'business_products'
        id_field = 'id' if not is_product else 'id'
        
        # FOR UPDATE获取行锁
        cursor.execute(
            f"SELECT stock FROM {table} WHERE {id_field} = %s FOR UPDATE",
            [sku_id]
        )
        
        result = cursor.fetchone()
        if not result:
            raise ValueError(f"商品不存在: {sku_id}")
        
        current_stock = result['stock']
        if current_stock < quantity:
            raise ValueError(f"库存不足: 当前{current_stock}, 需要{quantity}")
        
        # 扣减库存
        cursor.execute(
            f"UPDATE {table} SET stock = stock - %s WHERE {id_field} = %s",
            [quantity, sku_id]
        )
        
        conn.commit()
        
        logger.info(f"[OrderService] 库存锁定成功: sku_id={sku_id}, quantity={quantity}")
        return True
        
    except Exception as e:
        conn.rollback()
        logger.error(f"[OrderService] 库存锁定失败: {e}")
        raise
        
    finally:
        cursor.close()
        conn.close()
```

### 5.4 Redis完整集成

```python
# cache_redis.py

import redis
import json
from typing import Any, Optional

class RedisCache:
    """Redis缓存服务 - V35.0完整实现"""
    
    _instance = None
    _client = None
    
    def __new__(cls):
        if cls._instance is None:
            cls._instance = super().__new__(cls)
        return cls._instance
    
    def __init__(self):
        if self._client is None:
            self._client = redis.Redis(
                host=os.environ.get('REDIS_HOST', 'localhost'),
                port=int(os.environ.get('REDIS_PORT', 6379)),
                db=int(os.environ.get('REDIS_DB', 0)),
                password=os.environ.get('REDIS_PASSWORD'),
                decode_responses=True
            )
    
    def get(self, key: str) -> Optional[Any]:
        """获取缓存"""
        try:
            value = self._client.get(key)
            if value:
                return json.loads(value)
            return None
        except Exception as e:
            logger.error(f"Redis GET失败: {key}, {e}")
            return None
    
    def set(self, key: str, value: Any, ttl: int = 3600) -> bool:
        """设置缓存"""
        try:
            self._client.setex(key, ttl, json.dumps(value))
            return True
        except Exception as e:
            logger.error(f"Redis SET失败: {key}, {e}")
            return False
    
    def delete(self, key: str) -> bool:
        """删除缓存"""
        try:
            self._client.delete(key)
            return True
        except Exception as e:
            logger.error(f"Redis DELETE失败: {key}, {e}")
            return False
    
    # ============ 会话管理 ============
    
    def set_session(self, session_id: str, data: dict, ttl: int = 86400):
        """存储会话"""
        self.set(f"session:{session_id}", data, ttl)
    
    def get_session(self, session_id: str) -> Optional[dict]:
        """获取会话"""
        return self.get(f"session:{session_id}")
    
    def delete_session(self, session_id: str):
        """删除会话"""
        self.delete(f"session:{session_id}")
    
    # ============ 分布式限流 ============
    
    def rate_limit(self, key: str, limit: int, window: int) -> bool:
        """
        滑动窗口限流
        
        Args:
            key: 限流key
            limit: 限制次数
            window: 时间窗口（秒）
        
        Returns:
            True: 允许通过
            False: 被限流
        """
        try:
            current = self._client.get(f"rate:{key}")
            if current is None:
                self._client.setex(f"rate:{key}", window, 1)
                return True
            
            if int(current) >= limit:
                return False
            
            self._client.incr(f"rate:{key}")
            return True
        except Exception as e:
            logger.error(f"限流检查失败: {e}")
            return True  # 失败时放行
```
